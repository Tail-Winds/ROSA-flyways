centers <- lapply(
  locations,
  \(.) {
    .[, .(med_lon = median(lon, na.rm = T), med_lat = median(lat, na.rm = T))]
  }
)
proj_dist <- function(stations, centers) {
  list(
    aeqd = azeq_fwd(stations, centers[[1]], centers[[2]]) |>
      dist() |>
      as.matrix(),
    utm_auto = utmups_fwd(stations)[, 1:2] |>
      dist() |>
      as.matrix(),
    utm17 = tm_exact_fwd(stations, -81) |>
      dist() |>
      as.matrix(),
    utm18 = tm_exact_fwd(stations, -75) |>
      dist() |>
      as.matrix(),
    utm19 = tm_exact_fwd(stations, -69) |>
      dist() |>
      as.matrix(),
    tm = tm_exact_fwd(stations, centers[[1]]) |>
      dist() |>
      as.matrix(),
    cass = cassini_fwd(stations, centers[[1]], centers[[2]])[, 1:2] |>
      dist() |>
      as.matrix()
  )
}

projected_distances <- mapply(proj_dist, locations, centers, SIMPLIFY = F)
proj_errs <- function(dists, truth, species) {
  lapply(dists[[species]], \(.) {
    (. - truth[[species]])[lower.tri(truth[[species]])]
  })
}
